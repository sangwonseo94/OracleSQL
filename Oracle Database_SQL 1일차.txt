SELECT DEPTNO,GRADE, COUNT(*),ROUND(AVG(WEIGHT),1) 
FROM STUDENT
GROUP BY DEPTNO,GRADE
ORDER BY DEPTNO,GRADE;-- ROUND함수 (뭐를?,몇째자리에서?),몇째 자리에서를 안주면 소수점 1번째 자리에서 반올림하여서 보여준다.

SELECT DEPTNO, POSITION, COUNT(*)
FROM PROFESSOR
GROUP BY ROLLUP (DEPTNO, POSITION); -- ROLLUP은 전체행을 그룹화 하고 각그룹에 대해서 부분합을 구한다.

SELECT DEPTNO, POSITION, COUNT(*)
FROM PROFESSOR
GROUP BY CUBE (DEPTNO, POSITION); -- ROLLUP은 전체행을 그룹화 하고 각그룹에 대해서 부분합을 구한다.

SELECT GRADE,COUNT(*),Round(AVG(HEIGHT)) AVG_HEIGHT --한칸뛰면 그 이름으로 나온다.
from STUDENT
group by GRADE
having COUNT(*)>4
order by AVG_HEIGHT desc;
